## 1. 入口与配置 (cmd/api/main.go)

```go
// cmd/api/main.go
package main

import (
	"log"
	"os"
	"os/signal"
	"syscall"

	"github.com/gin-gonic/gin"
	"lunar-week-one/internal/config"
	"lunar-week-one/internal/handlers"
	"lunar-week-one/internal/middleware"
	"lunar-week-one/internal/repository"
	"lunar-week-one/internal/services"
)

func main() {
	// 加载配置
	cfg, err := config.Load()
	if err != nil {
		log.Fatalf("Failed to load config: %v", err)
	}

	// 初始化数据库
	db, err := repository.NewDB(cfg.DatabaseURL)
	if err != nil {
		log.Fatalf("Failed to connect to database: %v", err)
	}
	defer db.Close()

	// 初始化服务层
	sessionRepo := repository.NewSessionRepository(db)
	progressRepo := repository.NewProgressRepository(db)
	eventRepo := repository.NewEventRepository(db)
	
	sessionService := services.NewSessionService(sessionRepo)
	progressService := services.NewProgressService(progressRepo)
	analyticsService := services.NewAnalyticsService(eventRepo)

	// 初始化处理器
	sessionHandler := handlers.NewSessionHandler(sessionService)
	progressHandler := handlers.NewProgressHandler(progressService)
	eventHandler := handlers.NewEventHandler(eventRepo)
	configHandler := handlers.NewConfigHandler(cfg)

	// 创建Gin引擎
	r := gin.Default()
	
	// 添加CORS中间件
	r.Use(middleware.CORS())

	// 健康检查
	r.GET("/health", func(c *gin.Context) {
		c.JSON(200, gin.H{"status": "ok"})
	})

	// API路由
	v1 := r.Group("/api/v1")
	{
		// 会话管理
		v1.POST("/sessions", sessionHandler.Create)
		v1.GET("/sessions/:id", sessionHandler.Get)
		
		// 进度管理
		v1.PUT("/sessions/:id/progress", progressHandler.Update)
		v1.GET("/sessions/:id/progress", progressHandler.GetBySession)
		
		// 事件上报
		v1.POST("/sessions/:id/events", eventHandler.Submit)
		
		// 配置获取
		v1.GET("/config", configHandler.Get)
	}

	// 优雅关闭
	go func() {
		sigChan := make(chan os.Signal, 1)
		signal.Notify(sigChan, syscall.SIGINT, syscall.SIGTERM)
		<-sigChan
		log.Println("Shutting down gracefully...")
	}()

	// 启动服务器
	addr := ":" + cfg.ServerPort
	log.Printf("Server starting on %s", addr)
	if err := r.Run(addr); err != nil {
		log.Fatalf("Failed to start server: %v", err)
	}
}
```

```go
// internal/config/config.go
package config

import (
	"fmt"
	"os"
	"strconv"

	"github.com/joho/godotenv"
)

// Config 应用配置
type Config struct {
	ServerPort   string
	DatabaseURL  string
	RedisURL     string
	MinIOEndpoint string
	MinIOBucket  string
	Environment  string
	
	// 音频配置
	AudioBaseURL string
	
	// 内容配置
	ContentBaseURL string
	
	// 分析配置
	AnalyticsEnabled bool
}

const (
	defaultPort = "8080"
)

// Load 加载配置
func Load() (*Config, error) {
	// 加载.env文件
	if err := godotenv.Load(); err != nil {
		// 如果没有.env文件，使用环境变量
	}
	
	cfg := &Config{
		ServerPort: getEnvOrDefault("SERVER_PORT", defaultPort),
		DatabaseURL: getEnvOrDefault("DATABASE_URL", "postgres://localhost:5432/lunar_week_one?sslmode=disable"),
		RedisURL: getEnvOrDefault("REDIS_URL", "localhost:6379"),
		MinIOEndpoint: getEnvOrDefault("MINIO_ENDPOINT", "localhost:9000"),
		MinIOBucket: getEnvOrDefault("MINIO_BUCKET", "lunar-week-one"),
		Environment: getEnvOrDefault("ENVIRONMENT", "development"),
		AudioBaseURL: getEnvOrDefault("AUDIO_BASE_URL", "https://cdn.lunarweekone.com/audio/"),
		ContentBaseURL: getEnvOrDefault("CONTENT_BASE_URL", "https://cdn.lunarweekone.com/content/"),
		AnalyticsEnabled: getEnvAsBool("ANALYTICS_ENABLED", true),
	}
	
	return cfg, nil
}

// getEnvOrDefault 获取环境变量或默认值
func getEnvOrDefault(key, defaultValue string) string {
	if value := os.Getenv(key); value != "" {
		return value
	}
	return defaultValue
}

// getEnvAsBool 获取布尔类型的环境变量
func getEnvAsBool(key string, defaultVal bool) bool {
	value := os.Getenv(key)
	if value == "" {
		return defaultVal
	}
	result, err := strconv.ParseBool(value)
	if err != nil {
		return defaultVal
	}
	return result
}

// GetDSN 获取PostgreSQL连接字符串
func (c *Config) GetDSN() string {
	return fmt.Sprintf("host=%s port=5432 user=postgres password=postgres dbname=lunar_week_one sslmode=disable",
		c.DatabaseURL)
}
```

```go
	if j == nil {
		return nil, nil
	}
	return json.Marshal(j)

	if value == nil {
		*j = nil
		return nil
	}

		return errors.New("failed to scan JSONB")

	return json.Unmarshal(bytes, j)
```

```go
// internal/repository/db.go
package repository

import (
	"database/sql"
	"fmt"
	"time"

	_ "github.com/lib/pq"
	"lunar-week-one/internal/models"
)

// DB 数据库连接
type DB struct {
	*sql.DB
}

// NewDB 创建数据库连接
func NewDB(connStr string) (*DB, error) {
	db, err := sql.Open("postgres", connStr)
	if err != nil {
		return nil, fmt.Errorf("failed to open database: %w", err)
	}

	// 配置连接池
	db.SetMaxOpenConns(25)
	db.SetMaxIdleConns(5)
	db.SetConnMaxLifetime(5 * time.Minute)

	// 测试连接
	if err := db.Ping(); err != nil {
		return nil, fmt.Errorf("failed to ping database: %w", err)
	}

	return &DB{db}, nil
}

// InitSchema 初始化数据库schema
func (db *DB) InitSchema() error {
	schema := `
	-- 用户表
	CREATE TABLE IF NOT EXISTS users (
		id BIGSERIAL PRIMARY KEY,
		uuid UUID NOT NULL UNIQUE DEFAULT gen_random_uuid(),
		created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
		updated_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
		settings JSONB DEFAULT '{}'::JSONB,
		anonymous_id UUID
	);

	-- 会话表
	CREATE TABLE IF NOT EXISTS sessions (
		id BIGSERIAL PRIMARY KEY,
		uuid UUID NOT NULL DEFAULT gen_random_uuid(),
		user_id BIGINT REFERENCES users(id),
		started_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
		completed_at TIMESTAMP WITH TIME ZONE,
		duration_seconds INTEGER,
		mood_before INTEGER CHECK (mood_before BETWEEN 1 AND 5),
		mood_after INTEGER CHECK (mood_after BETWEEN 1 AND 5),
		completion_status VARCHAR(20) DEFAULT 'in_progress' 
		 CHECK (completion_status IN ('in_progress', 'completed', 'abandoned')),
		overall_rating INTEGER CHECK (overall_rating BETWEEN 1 AND 5),
		device_type VARCHAR(50),
		platform VARCHAR(20),
		created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
	);

	-- 进度表
	CREATE TABLE IF NOT EXISTS session_progress (
		id BIGSERIAL PRIMARY KEY,
		session_id BIGINT NOT NULL REFERENCES sessions(id),
		day_number INTEGER NOT NULL CHECK (day_number BETWEEN 1 AND 7),
		entered_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
		exited_at TIMESTAMP WITH TIME ZONE,
		status VARCHAR(20) DEFAULT 'not_started' 
		 CHECK (status IN ('not_started', 'in_progress', 'completed', 'skipped')),
		completion_rate DECIMAL(5,2) DEFAULT 0.00,
		rituals_completed JSONB DEFAULT '[]'::JSONB,
		interaction_count INTEGER DEFAULT 0,
		created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
		updated_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
		UNIQUE(session_id, day_number)
	);

	-- 事件表
	CREATE TABLE IF NOT EXISTS events (
		id BIGSERIAL PRIMARY KEY,
		id_uuid UUID NOT NULL DEFAULT gen_random_uuid(),
		session_id BIGINT NOT NULL REFERENCES sessions(id),
		event_type VARCHAR(100) NOT NULL,
		event_data JSONB DEFAULT '{}'::JSONB,
		occurred_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
	);

	-- 索引
	CREATE INDEX IF NOT EXISTS idx_sessions_user_id ON sessions(user_id);
	CREATE INDEX IF NOT EXISTS idx_sessions_uuid ON sessions(uuid);
	CREATE INDEX IF NOT EXISTS idx_sessions_started_at ON sessions(started_at);
	CREATE INDEX IF NOT EXISTS idx_progress_session_id ON session_progress(session_id);
	CREATE INDEX IF NOT EXISTS idx_events_session_id ON events(session_id);
	CREATE INDEX IF NOT EXISTS idx_events_occurred_at ON events(occurred_at);
	`

	_, err := db.Exec(schema)
	return err
}

// Transaction 执行事务
func (db *DB) Transaction(fn func(*sql.Tx) error) error {
	tx, err := db.Begin()
	if err != nil {
		return fmt.Errorf("failed to begin transaction: %w", err)
	}

	defer func() {
		if p := recover(); p != nil {
			tx.Rollback()
			panic(p)
		}
	}()

	if err := fn(tx); err != nil {
		tx.Rollback()
		return err
	}

	return tx.Commit()
}
```

```go
package repository

import (
	"database/sql"

// SessionRepository 会话数据访问

// NewSessionRepository 创建会话仓库
func NewSessionRepository(db *DB) *SessionRepository {
	return &SessionRepository{db: db}
}

// Create 创建新会话
func (r *SessionRepository) Create(session *models.Session) error {
	query := `
		INSERT INTO sessions (user_id, device_type, platform, completion_status)
		VALUES ($1, $2, $3, 'in_progress')
		RETURNING id, uuid, started_at, created_at
	`

	err := r.db.QueryRow(
		query,
		session.UserID,
		session.DeviceType,
		session.Platform,
	).Scan(
		&session.ID,
		&session.UUID,
		&session.StartedAt,
		&session.CreatedAt,
	)

	if err != nil {
		return fmt.Errorf("failed to create session: %w", err)
	}

	return nil
}

// GetByID 通过ID获取会话
```

```go
// BatchInsert 批量插入事件
func (r *EventRepository) BatchInsert(events []*models.Event) error {
	tx, err := r.db.Begin()
	if err != nil {
		return fmt.Errorf("failed to begin transaction: %w", err)
	}
	defer tx.Rollback()

	stmt, err := tx.Prepare(`
		INSERT INTO events (session_id, event_type, event_data, occurred_at)
		VALUES ($1, $2, $3, $4)
	`)
	if err != nil {
		return fmt.Errorf("failed to prepare statement: %w", err)
	}
	defer stmt.Close()

	for _, event := range events {
		occurredAt := event.OccurredAt
		if occurredAt.IsZero() {
			occurredAt = time.Now()
		}

		_, err := stmt.Exec(
			event.SessionID,
			event.EventType,
			event.EventData,
			occurredAt,
		)
		if err != nil {
			return fmt.Errorf("failed to insert event: %w", err)
		}
	}

	return tx.Commit()
}
```

```go
	SessionID        int64          `json:"session_id"`
	TotalEvents      int            `json:"total_events"`
	EventBreakdown   map[string]int `json:"event_breakdown"`
	RitualCompletion map[string]float64 `json:"ritual_completion"`
	Duration         time.Duration  `json:"duration"`
	EngagementScore  float64        `json:"engagement_score"`
}

// GetSessionAnalytics 获取会话分析
func (s *AnalyticsService) GetSessionAnalytics(sessionID int64) (*SessionAnalytics, error) {
	events, err := s.eventRepo.GetBySession(sessionID)
	if err != nil {
		return nil, err
	}

	analytics := &SessionAnalytics{
		SessionID:        sessionID,
		TotalEvents:      len(events),
		EventBreakdown:   make(map[string]int),
		RitualCompletion: make(map[string]float64),
	}

	var firstEvent, lastEvent time.Time

	for _, event := range events {
		// 统计事件类型
		analytics.EventBreakdown[event.EventType]++

		// 记录时间范围
		if firstEvent.IsZero() || event.OccurredAt.Before(firstEvent) {
			firstEvent = event.OccurredAt
		}
		if event.OccurredAt.After(lastEvent) {
			lastEvent = event.OccurredAt
		}

		// 计算仪式完成率
		if event.EventType == "ritual_completed" {
			if completion, ok := event.EventData["completion_rate"].(float64); ok {
				if ritualType, ok := event.EventData["ritual_type"].(string); ok {
					// 计算平均完成率
					existing := analytics.RitualCompletion[ritualType]
					count := float64(analytics.EventBreakdown["ritual_completed_"+ritualType])
					analytics.RitualCompletion[ritualType] = (existing*count + completion) / (count + 1)
					analytics.EventBreakdown["ritual_completed_"+ritualType]++
				}
			}
		}
	}

	if !firstEvent.IsZero() && !lastEvent.IsZero() {
		analytics.Duration = lastEvent.Sub(firstEvent)
	}

	// 计算参与度分数 (0-100)
	analytics.EngagementScore = s.calculateEngagementScore(analytics)

	return analytics, nil
}

// calculateEngagementScore 计算参与度分数
func (s *AnalyticsService) calculateEngagementScore(analytics *SessionAnalytics) float64 {
	// 基于事件数量和仪式完成率计算
	eventScore := float64(analytics.TotalEvents) / 30.0 // 假设30个事件为满分
	if eventScore > 1 {
		eventScore = 1
	}

	// 仪式完成率
	var avgCompletion float64
	if len(analytics.RitualCompletion) > 0 {
		for _, rate := range analytics.RitualCompletion {
			avgCompletion += rate
		}
		avgCompletion /= float64(len(analytics.RitualCompletion))
	}

	// 综合分数
	return (eventScore*0.4 + avgCompletion*0.6) * 100
}
```

```go
// @Tags events

// @Produce json
// @Param id path string true "会话ID"
// @Param request body models.EventSubmitRequest true "事件提交请求"

	// 获取会话ID
	// 这里简化处理，实际应该通过服务层验证
```

```csharp
    {
        var requestData = new Dictionary<string, object>
        {
            { "device_type", SystemInfo.deviceType.ToString() },
            { "platform", Application.platform.ToString() },
            { "client_version", Application.version },
            { "anonymous_id", PlayerPrefs.GetString("anonymous_id", "") }
        };
        
        var json = JsonUtility.ToJson(new SerializableDictionary(requestData));

            request.uploadHandler = new UploadHandlerRaw(System.Text.Encoding.UTF8.GetBytes(json));
            request.downloadHandler = new DownloadHandlerBuffer();
            request.SetRequestHeader("Content-Type", "application/json");
            
            yield return request.SendWebRequest();
            
            if (request.result == UnityWebRequest.Result.Success)
            {
                var response = JsonUtility.FromJson<SessionResponse>(request.downloadHandler.text);
                sessionId = response.data.session_uuid;
                PlayerPrefs.SetString("session_id", sessionId);
                onComplete?.Invoke(response);
            }
            else
            {
                Debug.LogError($"Failed to create session: {request.error}");
                // 使用本地会话ID
                sessionId = Guid.NewGuid().ToString();
                onComplete?.Invoke(null);
            }
        }
    }
    
    /// <summary>
    /// 更新进度
    /// </summary>
    public IEnumerator UpdateProgress(int dayNumber, ProgressUpdateRequest request, Action<bool> onComplete)
    {
        if (string.IsNullOrEmpty(sessionId))
        {
            // 使用本地存储
            SaveProgressLocally(dayNumber, request);
            onComplete?.Invoke(true);
            yield break;
        }
        
        var json = JsonUtility.ToJson(request);
        
        using (UnityWebRequest www = new UnityWebRequest($"{apiBaseUrl}/sessions/{sessionId}/progress", "PUT"))
        {
            www.uploadHandler = new UploadHandlerRaw(System.Text.Encoding.UTF8.GetBytes(json));
            www.downloadHandler = new DownloadHandlerBuffer();
            www.SetRequestHeader("Content-Type", "application/json");
            
            yield return www.SendWebRequest();
            
            if (www.result == UnityWebRequest.Result.Success)
            {
                onComplete?.Invoke(true);
            }
            else
            {
                // 降级到本地存储
                SaveProgressLocally(dayNumber, request);
                onComplete?.Invoke(true);
            }
        }
    }
```

```csharp
    {
        StartCoroutine(BreathingSyncRoutine(pattern));

    private IEnumerator BreathingSyncRoutine(BreathingPattern pattern)
    {
        float elapsed = 0f;
        float cycleDuration = 60f / pattern.BPM; // 一拍持续时间
        
        while (pattern.IsActive)
        {
            elapsed += Time.deltaTime;
            float phase = elapsed / cycleDuration;
            
            // 根据呼吸模式调整音量/音调
            float modulation = GetBreathingModulation(phase, pattern.Type);
            
            if (ritualSource != null)
            {
                ritualSource.pitch = 1f + modulation * 0.05f;
                ritualSource.volume = ritualVolume * masterVolume * (0.8f + modulation * 0.2f);
            }
            
            // 检测节拍
            if (phase >= 1f)
            {
                elapsed = 0f;
                OnBeatDetected?.Invoke(phase);
            }
            
            yield return null;
        }
        
        // 重置
        ritualSource.pitch = 1f;
        ritualSource.volume = ritualVolume * masterVolume;
    }
    
    private float GetBreathingModulation(float phase, BreathingType type)
    {
        switch (type)
        {
            case BreathingType.Box:
                // 4-7-8 呼吸模式
                if (phase < 0.25f) return Mathf.Lerp(0f, 1f, phase * 4); // 吸气
                if (phase < 0.5f) return Mathf.Lerp(1f, 0f, (phase - 0.25f) * 4); // 屏息
                if (phase < 0.75f) return Mathf.Lerp(0f, -1f, (phase - 0.5f) * 4); // 呼气
                return Mathf.Lerp(-1f, 0f, (phase - 0.75f) * 4); // 屏息
                
            case BreathingType.Coherent:
                // 一致性呼吸 - 均匀的吸气和呼气
                return Mathf.Sin(phase * Mathf.PI * 2) * 0.5f;
                
            case BreathingType.Resonance:
                // 共振呼吸 - 6次/分钟
                return Mathf.Sin(phase * Mathf.PI * 2) * 0.6f;
                
            default:
                return Mathf.Sin(phase * Mathf.PI * 2);
        }
    }
    
    /// <summary>
    /// 停止呼吸同步
    /// </summary>
    public void StopBreathingSync()
    {
        StopAllCoroutines();
        if (ritualSource != null)
        {
            ritualSource.pitch = 1f;
        }
    }
    
    // 淡入淡出辅助方法
    private void FadeTo(AudioSource source, AudioClip clip, float targetVolume)
    {
        StartCoroutine(FadeRoutine(source, clip, targetVolume));
    }
    
    private void FadeOut(AudioSource source)
    {
        StartCoroutine(FadeOutRoutine(source));
    }
    
    private IEnumerator FadeRoutine(AudioSource source, AudioClip clip, float targetVolume)
    {
        float startVolume = source.volume;
        source.clip = clip;
        source.Play();
        
        float elapsed = 0f;
        float duration = 2f;
        
        while (elapsed < duration)
        {
            elapsed += Time.deltaTime;
            source.volume = Mathf.Lerp(startVolume, targetVolume, elapsed / duration);
            yield return null;
        }
        
        source.volume = targetVolume;
    }
    
    private IEnumerator FadeOutRoutine(AudioSource source)
    {
        float startVolume = source.volume;
        float elapsed = 0f;
        float duration = 2f;
        
        while (elapsed < duration)
        {
            elapsed += Time.deltaTime;
            source.volume = Mathf.Lerp(startVolume, 0f, elapsed / duration);
            yield return null;
        }
        
        source.Stop();
        source.volume = startVolume;
    }
    
    // 资源加载
    private AudioClip LoadClip(string path)
    {
        if (audioCache.TryGetValue(path, out var cached))
        {
            return cached;
        }
        
        var clip = Resources.Load<AudioClip>(path);
        if (clip != null)
        {
            audioCache[path] = clip;
        }
        return clip;
    }
    
    /// <summary>
    /// 清理资源
    /// </summary>
    public void Dispose()
    {
        StopAll();
        audioCache.Clear();
    }
}
```

```csharp
        phase.OnEnter?.Invoke();

            audioManager.PlayRitual(phase.AudioCue, phase.LoopAudio);

        while (elapsed < duration && currentSession.IsActive)
        {
            elapsed += Time.deltaTime;
            
            // 更新进度
            float phaseProgress = elapsed / duration;
            float totalProgress = currentSession.Progress + phaseProgress * (duration / config.DurationSec);
            currentSession.ElapsedTime = elapsed;
            currentSession.Progress = totalProgress;
            
            OnProgressUpdate?.Invoke(totalProgress);
            
            // 检查用户交互
            if (CheckUserInteraction(phase))
            {
                HandleInteraction(phase);
            }
            
            // 检查呼吸同步
            if (phase.RequiresBreathSync)
            {
                var breathData = ProcessBreathingInput();
                OnBreathingFeedback?.Invoke(breathData.SyncRate);
            }
            
            yield return null;
        }
        
        // 阶段结束
        phase.OnExit?.Invoke();
        
        if (!string.IsNullOrEmpty(phase.AudioCue))
        {
            audioManager.StopRitual();
        }
    }

            routine = null;

        audioManager.StopBreathingSync();

                { "ritual_type", currentSession.Config.Type },
                { "progress", currentSession.Progress },
```

```csharp
// Assets/Scripts/Progress/ProgressManager.cs
using System;
using System.Collections.Generic;

/// <summary>
/// 进度管理器 - 管理用户在7天中的进度
/// </summary>
public class ProgressManager
{
    // 每日进度数据
    private Dictionary<int, DayProgress> dayProgress = new Dictionary<int, DayProgress>();
    
    // 当前状态
    public int CurrentDay => GetCurrentDay();
    public int CompletedDaysCount { get; private set; }
    public float OverallProgress { get; private set; }
    
    // 事件
    public event Action<int> OnDayComplete;
    public event Action<int, float> OnDayProgressUpdate;
    public event Action<float> OnOverallProgressUpdate;
    
    public ProgressManager()
    {
        Initialize();
    }
    
    private void Initialize()
    {
        // 初始化7天进度
        for (int i = 1; i <= 7; i++)
        {
            dayProgress[i] = new DayProgress
            {
                DayNumber = i,
                Status = DayStatus.Locked,
                CompletionRate = 0f,
                RitualsCompleted = new List<string>()
            };
        }
        
        // 解锁第1天
        dayProgress[1].Status = DayStatus.Available;
        CompletedDaysCount = 0;
        OverallProgress = 0f;
    }
    
    /// <summary>
    /// 进入某一天
    /// </summary>
    public void EnterDay(int day)
    {
        if (!dayProgress.ContainsKey(day)) return;
        if (dayProgress[day].Status == DayStatus.Locked) return;
        
        dayProgress[day].Status = DayStatus.InProgress;
        dayProgress[day].EnteredAt = DateTime.UtcNow;
        
        OnDayProgressUpdate?.Invoke(day, dayProgress[day].CompletionRate);
    }
    
    /// <summary>
    /// 完成某一天
    /// </summary>
    public void CompleteDay(int day)
    {
        if (!dayProgress.ContainsKey(day)) return;
        
        var progress = dayProgress[day];
        progress.Status = DayStatus.Completed;
        progress.CompletedAt = DateTime.UtcNow;
        progress.CompletionRate = 100f;
        
        // 解锁下一天
        if (day < 7)
        {
            dayProgress[day + 1].Status = DayStatus.Available;
        }
        
        CompletedDaysCount++;
        UpdateOverallProgress();
        
        OnDayComplete?.Invoke(day);
        OnDayProgressUpdate?.Invoke(day, 100f);
    }
    
    /// <summary>
    /// 记录仪式完成
    /// </summary>
    public void RecordRitualCompletion(int day, string ritualType, float completionRate)
    {
        if (!dayProgress.ContainsKey(day)) return;
        
        var progress = dayProgress[day];
        if (!progress.RitualsCompleted.Contains(ritualType))
        {
            progress.RitualsCompleted.Add(ritualType);
        }
        
        // 计算该天完成度
        progress.CompletionRate = CalculateDayCompletion(progress);
        UpdateOverallProgress();
        
        OnDayProgressUpdate?.Invoke(day, progress.CompletionRate);
    }
    
    /// <summary>
    /// 获取某天完成度
    /// </summary>
    public float GetDayCompletionRate(int day)
    {
        return dayProgress.ContainsKey(day) ? dayProgress[day].CompletionRate : 0f;
    }
    
    /// <summary>
    /// 获取当前天
    /// </summary>
    private int GetCurrentDay()
    {
        for (int i = 1; i <= 7; i++)
        {
            if (dayProgress[i].Status == DayStatus.InProgress)
            {
                return i;
            }
        }
        // 如果没有in_progress，返回第一个未完成的available
        for (int i = 1; i <= 7; i++)
        {
            if (dayProgress[i].Status == DayStatus.Available)
            {
                return i;
            }
        }
        return 1;
    }
    
    /// <summary>
    /// 计算当天完成度
    /// </summary>
    private float CalculateDayCompletion(DayProgress progress)
    {
        // 每个仪式贡献一定的完成度
        float ritualWeight = 100f / 3f; // 每个仪式1/3
        float completion = 0f;
        
        foreach (var ritual in progress.RitualsCompleted)
        {
            completion += ritualWeight;
        }
        
        return Mathf.Clamp(completion, 0f, 100f);
    }
    
    /// <summary>
    /// 更新整体完成度
    /// </summary>
    private void UpdateOverallProgress()
    {
        float total = 0f;
        for (int i = 1; i <= 7; i++)
        {
            total += dayProgress[i].CompletionRate;
        }
        OverallProgress = total / 7f;
        
        OnOverallProgressUpdate?.Invoke(OverallProgress);
    }
    
    /// <summary>
    /// 重置进度
    /// </summary>
    public void Reset()
    {
        Initialize();
    }
    
    /// <summary>
    /// 获取所有进度数据
    /// </summary>
    public Dictionary<int, DayProgress> GetAllProgress()
    {
        return new Dictionary<int, DayProgress>(dayProgress);
    }
}

/// <summary>
```

## 6. ResourceManager (资源管理)

```csharp
// Assets/Scripts/Resource/ResourceManager.cs
using System;

/// <summary>
/// 资源管理器 - 管理基地资源（能源、氧气、水）
/// </summary>
public class ResourceManager
{
    // 资源状态
    public ResourceStates CurrentState { get; private set; }
    
    // 资源衰减配置
    private ResourceDecayConfig decayConfig;
    
    // 状态
    public bool IsPaused { get; private set; }
    public event Action<ResourceStates> OnResourcesChanged;
    public event Action<ResourceStates> OnResourceWarning;
    public event Action<ResourceStates> OnResourceCritical;
    
    public ResourceManager()
    {
        ResetResources();
        LoadDecayConfig();
    }
    
    /// <summary>
    /// 重置资源到初始状态
    /// </summary>
    public void ResetResources()
    {
        CurrentState = new ResourceStates
        {
            Energy = 0.8f,
            Oxygen = 0.8f,
            Water = 0.8f
        };
        
        OnResourcesChanged?.Invoke(CurrentState);
    }
    
    /// <summary>
    /// 加载衰减配置
    /// </summary>
    private void LoadDecayConfig()
    {
        decayConfig = new ResourceDecayConfig
        {
            EnergyDecayPerMinute = 0.02f,
            OxygenDecayPerMinute = 0.015f,
            WaterDecayPerMinute = 0.01f,
            WarningThreshold = 0.3f,
            CriticalThreshold = 0.15f
        };
    }
    
    /// <summary>
    /// 暂停衰减（在仪式中）
    /// </summary>
    public void PauseResourceDecay()
    {
        IsPaused = true;
    }
    
    /// <summary>
    /// 恢复衰减
    /// </summary>
    public void ResumeResourceDecay()
    {
        IsPaused = false;
    }
    
    /// <summary>
    /// 更新资源（每帧调用）
    /// </summary>
    public void UpdateResources(float deltaTime)
    {
        if (IsPaused) return;
        
        float timeMultiplier = deltaTime / 60f; // 转换为每分钟
        
        // 衰减资源
        CurrentState.Energy -= decayConfig.EnergyDecayPerMinute * timeMultiplier;
        CurrentState.Oxygen -= decayConfig.OxygenDecayPerMinute * timeMultiplier;
        CurrentState.Water -= decayConfig.WaterDecayPerMinute * timeMultiplier;
        
        // 钳制在有效范围
        CurrentState.Energy = Mathf.Clamp01(CurrentState.Energy);
        CurrentState.Oxygen = Mathf.Clamp01(CurrentState.Oxygen);
        CurrentState.Water = Mathf.Clamp01(CurrentState.Water);
        
        // 检查阈值
        CheckThresholds();
    }
    
    /// <summary>
    /// 检查阈值并触发事件
    /// </summary>
    private void CheckThresholds()
    {
        float minResource = Mathf.Min(CurrentState.Energy, CurrentState.Oxygen, CurrentState.Water);
        
        if (minResource <= decayConfig.CriticalThreshold)
        {
            OnResourceCritical?.Invoke(CurrentState);
        }
        else if (minResource <= decayConfig.WarningThreshold)
        {
            OnResourceWarning?.Invoke(CurrentState);
        }
    }
    
    /// <summary>
    /// 调整能源
    /// </summary>
    public void AdjustEnergy(float amount)
    {
        CurrentState.Energy = Mathf.Clamp01(CurrentState.Energy + amount);
        OnResourcesChanged?.Invoke(CurrentState);
    }
    
    /// <summary>
    /// 调整氧气
    /// </summary>
    public void AdjustOxygen(float amount)
    {
        CurrentState.Oxygen = Mathf.Clamp01(CurrentState.Oxygen + amount);
        OnResourcesChanged?.Invoke(CurrentState);
    }
    
    /// <summary>
    /// 调整水
    /// </summary>
    public void AdjustWater(float amount)
    {
        CurrentState.Water = Mathf.Clamp01(CurrentState.Water + amount);
        OnResourcesChanged?.Invoke(CurrentState);
    }
    
    /// <summary>
    /// 执行资源维护操作
    /// </summary>
    public void PerformMaintenance(MaintenanceType type)
    {
        float recovery = 0f;
        
        switch (type)
        {
            case MaintenanceType.AdjustSolarPanels:
                recovery = 0.1f;
                AdjustEnergy(recovery);
                break;
            case MaintenanceType.CheckOxygen:
                recovery = 0.08f;
                AdjustOxygen(recovery);
                break;
            case MaintenanceType.CheckWater:
                recovery = 0.05f;
                AdjustWater(recovery);
                break;
            case MaintenanceType.FullMaintenance:
                AdjustEnergy(0.1f);
                AdjustOxygen(0.1f);
                AdjustWater(0.1f);
                break;
        }
    }
    
    /// <summary>
    /// 获取资源状态描述
    /// </summary>
    public ResourceStatus GetResourceStatus()
    {
        return new ResourceStatus
        {
            Energy = GetLevelStatus(CurrentState.Energy),
            Oxygen = GetLevelStatus(CurrentState.Oxygen),
            Water = GetLevelStatus(CurrentState.Water)
        };
    }
    
    private LevelStatus GetLevelStatus(float value)
    {
        if (value <= decayConfig.CriticalThreshold) return LevelStatus.Critical;
        if (value <= decayConfig.WarningThreshold) return LevelStatus.Warning;
        return LevelStatus.Normal;
    }
}
```

## 7. EventManager (事件追踪)

```csharp
// Assets/Scripts/Core/EventManager.cs
using System;
using System.Collections;
using System.Collections.Generic;
using UnityEngine;

/// <summary>
/// 事件管理器 - 收集和上报用户行为事件
/// </summary>
public class EventManager
{
    private DataManager dataManager;
    private List<EventData> eventQueue = new List<EventData>();
    private bool isProcessing = false;
    
    // 批处理配置
    private float batchInterval = 5f;
    private int batchSize = 10;
    private float lastFlushTime;
    
    public EventManager()
    {
        dataManager = GameManager.Instance.DataManager;
        lastFlushTime = Time.time;
    }
    
    /// <summary>
    /// 追踪事件
    /// </summary>
    public void TrackEvent(string eventType, Dictionary<string, object> properties)
    {
        var evt = new EventData
        {
            EventType = eventType,
            Properties = properties,
            Timestamp = DateTime.UtcNow,
            SessionID = GameManager.Instance.CurrentSession?.SessionID ?? ""
        };
        
        // 添加到队列
        eventQueue.Add(evt);
        
        // 立即处理关键事件
        if (IsCriticalEvent(eventType))
        {
            ProcessEventImmediate(evt);
        }
        
        // 检查是否需要批量处理
        if (eventQueue.Count >= batchSize || Time.time - lastFlushTime > batchInterval)
        {
            FlushEvents();
        }
    }
    
    /// <summary>
    /// 判断是否为关键事件
    /// </summary>
    private bool IsCriticalEvent(string eventType)
    {
        return eventType == "session_started" ||
               eventType == "session_ended" ||
               eventType == "day_completed" ||
               eventType == "ritual_completed";
    }
    
    /// <summary>
    /// 立即处理事件
    /// </summary>
    private void ProcessEventImmediate(EventData evt)
    {
        CoroutineManager.Start(ProcessEventRoutine(evt));
    }
    
    /// <summary>
    /// 批量刷新事件
    /// </summary>
    public void FlushEvents()
    {
        if (eventQueue.Count == 0 || isProcessing) return;
        
        var eventsToProcess = new List<EventData>(eventQueue);
        eventQueue.Clear();
        lastFlushTime = Time.time;
        
        CoroutineManager.Start(ProcessBatchRoutine(eventsToProcess));
    }
    
    /// <summary>
    /// 处理单个事件
    /// </summary>
    private IEnumerator ProcessEventRoutine(EventData evt)
    {
        var properties = new Dictionary<string, object>
        {
            { "event_type", evt.EventType },
            { "session_id", evt.SessionID },
            { "timestamp", evt.Timestamp.ToString("o") }
        };
        
        foreach (var kvp in evt.Properties)
        {
            properties[kvp.Key] = kvp.Value;
        }
        
        yield return dataManager.TrackEvent(evt.EventType, properties);
    }
    
    /// <summary>
    /// 批量处理事件
    /// </summary>
    private IEnumerator ProcessBatchRoutine(List<EventData> events)
    {
        isProcessing = true;
        
        // 分批处理
        for (int i = 0; i < events.Count; i += batchSize)
        {
            var batch = events.GetRange(i, Math.Min(batchSize, events.Count - i));
            
            // 构建批量请求
            var batchData = new BatchEventRequest
            {
                session_id = GameManager.Instance.CurrentSession?.SessionID ?? "",
                events = batch
            };
            
            // 发送请求（需要实现）
            // yield return SendBatchRequest(batchData);
            
            yield return new WaitForSeconds(0.1f);
        }
        
        isProcessing = false;
    }
    
    /// <summary>
    /// 记录体验前情绪
    /// </summary>
    public void RecordMoodBefore(int mood)
    {
        TrackEvent("mood_before", new Dictionary<string, object>
        {
            { "mood_score", mood }
        });
    }
    
    /// <summary>
    /// 记录体验后情绪
    /// </summary>
    public void RecordMoodAfter(int mood)
    {
        TrackEvent("mood_after", new Dictionary<string, object>
        {
            { "mood_score", mood }
        });
    }
    
    /// <summary>
    /// 记录整体评分
    /// </summary>
    public void RecordOverallRating(int rating)
    {
        TrackEvent("overall_rating", new Dictionary<string, object>
        {
            { "rating", rating }
        });
    }
    
    /// <summary>
    /// 清理
    /// </summary>
    public void Dispose()
    {
        FlushEvents();
    }
}

/// <summary>
/// 事件数据
/// </summary>
public class EventData
{
    public string EventType;
    public Dictionary<string, object> Properties;
    public DateTime Timestamp;
    public string SessionID;
}

/// <summary>
/// 批量事件请求
/// </summary>
[Serializable]
public class BatchEventRequest
{
    public string session_id;
    public List<EventData> events;
}
```
