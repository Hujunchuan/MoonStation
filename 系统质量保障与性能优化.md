    require.NoError(t, err)

func setupTestRouter(db *repository.DB) *gin.Engine {
    gin.SetMode(gin.TestMode)
    r := gin.New()
    
    // 初始化服务
    sessionRepo := repository.NewSessionRepository(db)

func TestAPI_Integration(t *testing.T) {
    if testing.Short() {
        t.Skip("Skipping integration test in short mode")
    }
    
    // Setup
    pgContainer, db := setupTestContainer(t)
    defer func() {
        if err := pgContainer.Terminate(context.Background()); err != nil {
            t.Logf("Failed to terminate container: %v", err)
        }
    }()
    
    testRouter = setupTestRouter(db)
    
    t.Run("完整会话流程", func(t *testing.T) {
        // 1. 创建会话

        assert.True(t, createResp.Success)

        sessionID := sessionData["session_uuid"].(string)
        
        req, _ = http.NewRequest("GET", "/api/v1/sessions/"+sessionID, nil)
        w = httptest.NewRecorder()
        testRouter.ServeHTTP(w, req)
        
        assert.Equal(t, http.StatusOK, w.Code)
        
        // 3. 更新进度
        progressReq := models.ProgressUpdateRequest{
            DayNumber:        1,
            Action:           "progress",
            DurationSeconds:  180,
            CompletionRate:   75.0,
            RitualsCompleted: []string{"breathing"},
        }
        jsonBody, _ = json.Marshal(progressReq)
        
        req, _ = http.NewRequest("PUT", "/api/v1/sessions/"+sessionID+"/progress", bytes.NewBuffer(jsonBody))
        req.Header.Set("Content-Type", "application/json")
        w = httptest.NewRecorder()
        testRouter.ServeHTTP(w, req)
        
        assert.Equal(t, http.StatusOK, w.Code)
        
        var progressResp models.APIResponse
        json.Unmarshal(w.Body.Bytes(), &progressResp)
        assert.True(t, progressResp.Success)
    })
    
    t.Run("并发请求处理", func(t *testing.T) {
        // 创建多个会话的并发测试
        numRequests := 10
        done := make(chan bool, numRequests)
        
        for i := 0; i < numRequests; i++ {
            go func(idx int) {
                sessionReq := CreateRequest{
                    DeviceType: "PC",
                    Platform:   "Windows",
                }
                jsonBody, _ := json.Marshal(sessionReq)
                
                req, _ := http.NewRequest("POST", "/api/v1/sessions", bytes.NewBuffer(jsonBody))
                req.Header.Set("Content-Type", "application/json")
                w := httptest.NewRecorder()
                testRouter.ServeHTTP(w, req)
                
                if w.Code == http.StatusCreated {
                    done <- true
                } else {
                    done <- false
                }
            }(i)
        }
        
        successCount := 0
        for i := 0; i < numRequests; i++ {
            if <-done {
                successCount++
            }
        }
        
        assert.Equal(t, numRequests, successCount, "All concurrent requests should succeed")
    })

import (

// Cache 缓存接口
type Cache interface {
    Get(ctx context.Context, key string) (string, error)
    Set(ctx context.Context, key string, value interface{}, expiration time.Duration) error
    Del(ctx context.Context, keys ...string) error
    GetOrSet(ctx context.Context, key string, fetchFunc func() (string, error), expiration time.Duration) (string, error)
    Incr(ctx context.Context, key string) (int64, error)
    Expire(ctx context.Context, key string, expiration time.Duration) error
}

// RedisCache Redis缓存实现
type RedisCache struct {
    client    *redis.Client
    prefix    string
}

// NewRedisCache 创建Redis缓存
func NewRedisCache(cfg *config.Config) (*RedisCache, error) {
    client := redis.NewClient(&redis.Options{
        Addr:         cfg.RedisURL,
        Password:     "", // 无密码
        DB:           0,
        PoolSize:     100,
        MinIdleConns: 10,
        DialTimeout:  5 * time.Second,
        ReadTimeout:  3 * time.Second,
        WriteTimeout: 3 * time.Second,
        PoolTimeout:  10 * time.Second,
    })
    
    ctx, cancel := context.WithTimeout(context.Background(), 5*time.Second)
    defer cancel()
    
    if err := client.Ping(ctx).Err(); err != nil {
        return nil, fmt.Errorf("failed to connect to redis: %w", err)
    }
    
    return &RedisCache{
        client: client,
        prefix: "lunar_week_one:",
    }, nil
}

// Get 获取缓存
func (c *RedisCache) Get(ctx context.Context, key string) (string, error) {
    return c.client.Get(ctx, c.prefix+key).Result()
}

// Set 设置缓存
func (c *RedisCache) Set(ctx context.Context, key string, value interface{}, expiration time.Duration) error {
    var strValue string
    switch v := value.(type) {
    case string:
        strValue = v
    default:
        data, err := json.Marshal(value)
        if err != nil {
            return fmt.Errorf("failed to marshal value: %w", err)
        }
        strValue = string(data)
    }
    
    return c.client.Set(ctx, c.prefix+key, strValue, expiration).Err()
}

// Del 删除缓存
func (c *RedisCache) Del(ctx context.Context, keys ...string) error {
    prefixedKeys := make([]string, len(keys))
    for i, key := range keys {
        prefixedKeys[i] = c.prefix + key
    }
    return c.client.Del(ctx, prefixedKeys...).Err()
}

// GetOrSet 获取或设置缓存
func (c *RedisCache) GetOrSet(ctx context.Context, key string, fetchFunc func() (string, error), expiration time.Duration) (string, error) {
    // 尝试获取缓存
    value, err := c.Get(ctx, key)
    if err == nil {
        return value, nil
    }
    
    // 缓存未命中，获取数据
    if !isCacheMiss(err) {
        return "", err
    }
    
    value, err = fetchFunc()
    if err != nil {
        return "", err
    }
    
    // 设置缓存
    if err := c.Set(ctx, key, value, expiration); err != nil {
        // 缓存设置失败不应影响主流程
        return value, nil
    }
    
    return value, nil
}

// Incr 递增
func (c *RedisCache) Incr(ctx context.Context, key string) (int64, error) {
    return c.client.Incr(ctx, c.prefix+key).Result()
}

// Expire 设置过期时间
func (c *RedisCache) Expire(ctx context.Context, key string, expiration time.Duration) error {
    return c.client.Expire(ctx, c.prefix+key, expiration).Err()
}

// isCacheMiss 判断是否为缓存未命中
func isCacheMiss(err error) bool {
    return err == redis.Nil
}

// Close 关闭连接
func (c *RedisCache) Close() error {
    return c.client.Close()
}

// SessionCache 会话缓存
type SessionCache struct {
    cache *RedisCache
}

// NewSessionCache 创建会话缓存
func NewSessionCache(cache *RedisCache) *SessionCache {
    return &SessionCache{cache: cache}
}

// GetSession 获取会话缓存
func (sc *SessionCache) GetSession(ctx context.Context, sessionID string) (map[string]interface{}, error) {
    key := fmt.Sprintf("session:%s", sessionID)
    data, err := sc.cache.Get(ctx, key)
    if err != nil {
        return nil, err
    }
    
    var result map[string]interface{}
    if err := json.Unmarshal([]byte(data), &result); err != nil {
        return nil, err
    }
    
    return result, nil
}

// SetSession 设置会话缓存
func (sc *SessionCache) SetSession(ctx context.Context, sessionID string, data map[string]interface{}) error {
    key := fmt.Sprintf("session:%s", sessionID)
    return sc.cache.Set(ctx, key, data, 24*time.Hour)
}

// InvalidateSession 使会话缓存失效
func (sc *SessionCache) InvalidateSession(ctx context.Context, sessionID string) error {
    key := fmt.Sprintf("session:%s", sessionID)
    return sc.cache.Del(ctx, key)
}

// GetMultipleSessionAnalytics 并行获取多个会话分析
func (s *ConcurrentAnalytics) GetMultipleSessionAnalytics(ctx context.Context, sessionIDs []string) ([]*SessionAnalyticsResult, error) {
    results := make([]*SessionAnalyticsResult, len(sessionIDs))
    
    // 使用工作池模式
    numWorkers := 10
    jobs := make(chan string, len(sessionIDs))
    resultsChan := make(chan *SessionAnalyticsResult, len(sessionIDs))
    
    // 启动工作池
    var wg sync.WaitGroup
    for i := 0; i < numWorkers; i++ {
        wg.Add(1)
        go func() {
            defer wg.Done()
            for sessionID := range jobs {
                result := s.analyzeSession(ctx, sessionID)
                resultsChan <- result
            }
        }()
    }
    
    // 发送任务
    go func() {
        for _, sessionID := range sessionIDs {
            jobs <- sessionID
        }
        close(jobs)
    }()
    
    // 收集结果
    go func() {
        wg.Wait()
        close(resultsChan)
    }()
    
    // 等待并收集结果
    for result := range resultsChan {
        results = append(results, result)
    }
    
    return results, nil
}

using UnityEngine;

/// <summary>
/// 对象池管理器 - 优化内存和性能
/// </summary>
public class ObjectPool : MonoBehaviour
{
    private static ObjectPool _instance;
    private Dictionary<string, Queue<GameObject>> _pools = new Dictionary<string, Queue<GameObject>>();
    private Dictionary<string, GameObject> _prefabs = new Dictionary<string, GameObject>();
    private Dictionary<GameObject, string> _poolMapping = new Dictionary<GameObject, string>();
    
    public static ObjectPool Instance
    {
        get
        {
            if (_instance == null)
            {
                var go = new GameObject("ObjectPool");
                _instance = go.AddComponent<ObjectPool>();
                DontDestroyOnLoad(go);
            }
            return _instance;
        }
    }
    
    /// <summary>
    /// 注册预制体到对象池
    /// </summary>
    public void RegisterPrefab(string poolId, GameObject prefab, int initialSize = 10)
    {
        if (_pools.ContainsKey(poolId))
        {
            Debug.LogWarning($"Pool {poolId} already exists");
            return;
        }
        
        _prefabs[poolId] = prefab;
        _pools[poolId] = new Queue<GameObject>();
        
        // 预热对象池
        WarmPool(poolId, initialSize);
    }
    
    /// <summary>
    /// 预热对象池
    /// </summary>
    public void WarmPool(string poolId, int count)
    {
        if (!_prefabs.ContainsKey(poolId))
        {
            Debug.LogError($"Prefab {poolId} not registered");
            return;
        }
        
        var prefab = _prefabs[poolId];
        var pool = _pools[poolId];
        
        for (int i = 0; i < count; i++)
        {
            var obj = Instantiate(prefab, transform);
            obj.SetActive(false);
            pool.Enqueue(obj);
            _poolMapping[obj] = poolId;
        }
    }
    
    /// <summary>
    /// 从对象池获取对象
    /// </summary>
    public GameObject Get(string poolId, Vector3 position = default, Quaternion rotation = default)
    {
        GameObject obj;
        
        if (_pools.TryGetValue(poolId, out var pool) && pool.Count > 0)
        {
            obj = pool.Dequeue();
        }
        else if (_prefabs.TryGetValue(poolId, out var prefab))
        {
            // 对象池为空，动态创建
            obj = Instantiate(prefab, transform);
            _poolMapping[obj] = poolId;
        }
        else
        {
            Debug.LogError($"Pool {poolId} not found");
            return null;
        }
        
        // 设置位置和旋转
        obj.transform.SetPositionAndRotation(position, rotation);
        obj.SetActive(true);
        
        return obj;
    }
    
    /// <summary>
    /// 归还对象到对象池
    /// </summary>
    public void Return(GameObject obj)
    {
        if (!_poolMapping.TryGetValue(obj, out var poolId))
        {
            Debug.LogWarning("Object not from pool, destroying");
            Destroy(obj);
            return;
        }
        
        obj.SetActive(false);
        _pools[poolId].Enqueue(obj);
    }
    
    /// <summary>
    /// 清空对象池
    /// </summary>
    public void ClearPool(string poolId)
    {
        if (!_pools.TryGetValue(poolId, out var pool)) return;
        
        while (pool.Count > 0)
        {
            var obj = pool.Dequeue();
            _poolMapping.Remove(obj);
            Destroy(obj);
        }
    }
    
    /// <summary>
    /// 清空所有对象池
    /// </summary>
    public void ClearAll()
    {
        foreach (var pool in _pools.Values)
        {
            while (pool.Count > 0)
            {
                var obj = pool.Dequeue();
                Destroy(obj);
            }
        }
        _poolMapping.Clear();
        _pools.Clear();
        _prefabs.Clear();
    }
}

using System;

/// <summary>
/// 内存优化管理器 - 监控和优化内存使用
/// </summary>
public class MemoryManager : MonoBehaviour
{
    public static MemoryManager Instance { get; private set; }
    
    // 内存阈值配置
    private const float WARNING_MEMORY_THRESHOLD = 0.8f; // 80%
    private const float CRITICAL_MEMORY_THRESHOLD = 0.9f; // 90%
    
    // 内存统计
    public long TotalAllocated { get; private set; }
    public long TotalReserved { get; private set; }
    public long MonoUsed { get; private set; }
    public float MemoryUsage { get; private set; }
    
    // 事件
    public event Action<float> OnMemoryWarning;
    public event Action<float> OnMemoryCritical;
    public event Action OnMemoryCleanup;
    
    // 资源追踪
    private Dictionary<string, int> _resourceReferenceCount = new Dictionary<string, int>();
    private List<WeakReference<Texture>> _textures = new List<WeakReference<Texture>>();
    
    private void Awake()
    {
        if (Instance == null)
        {
            Instance = this;
            DontDestroyOnLoad(gameObject);
        }
        else
        {
            Destroy(gameObject);
        }
    }
    
    private void Update()
    {
        UpdateMemoryStats();
        CheckMemoryThresholds();
    }
    
    /// <summary>
    /// 更新内存统计
    /// </summary>
    private void UpdateMemoryStats()
    {
        TotalAllocated = Profiler.GetTotalAllocatedMemoryLong();
        TotalReserved = Profiler.GetTotalReservedMemoryLong();
        MonoUsed = Profiler.GetMonoUsedMemoryLong();
        
        // 计算内存使用率
        long limit = Profiler.GetMemoryLimit();
        if (limit > 0)
        {
            MemoryUsage = (float)TotalAllocated / limit;
        }
    }
    
    /// <summary>
    /// 检查内存阈值
    /// </summary>
    private void CheckMemoryThresholds()
    {
        if (MemoryUsage >= CRITICAL_MEMORY_THRESHOLD)
        {
            OnMemoryCritical?.Invoke(MemoryUsage);
            PerformAggressiveCleanup();
        }
        else if (MemoryUsage >= WARNING_MEMORY_THRESHOLD)
        {
            OnMemoryWarning?.Invoke(MemoryUsage);
            PerformStandardCleanup();
        }
    }
    
    /// <summary>
    /// 标准清理
    /// </summary>
    public void PerformStandardCleanup()
    {
        // 清理未使用的资源
        Resources.UnloadUnusedAssets();
        
        // 清理空引用
        GC.Collect();
        
        OnMemoryCleanup?.Invoke();
    }
    
    /// <summary>
    /// 激进清理
    /// </summary>
    public void PerformAggressiveCleanup()
    {
        // 标准清理
        PerformStandardCleanup();
        
        // 强制清理 Mono 堆
        GC.Collect(0, GCCollectionMode.Forced);
        GC.WaitForPendingFinalizers();
        GC.Collect(0, GCCollectionMode.Forced);
        
        // 清理纹理缓存
        CleanupTextureCache();
        
        // 清理对象池
        ObjectPool.Instance.ClearPool("particles");
        ObjectPool.Instance.ClearPool("effects");
    }
    
    /// <summary>
    /// 清理纹理缓存
    /// </summary>
    private void CleanupTextureCache()
    {
        for (int i = _textures.Count - 1; i >= 0; i--)
        {
            if (_textures[i].TryGetTarget(out var texture))
            {
                if (texture != null)
                {
                    Resources.UnloadAsset(texture);
                }
            }
            else
            {
                _textures.RemoveAt(i);
            }
        }
    }
    
    /// <summary>
    /// 注册资源引用
    /// </summary>
    public void RegisterResourceReference(string resourceId)
    {
        if (!_resourceReferenceCount.ContainsKey(resourceId))
        {
            _resourceReferenceCount[resourceId] = 0;
        }
        _resourceReferenceCount[resourceId]++;
    }
    
    /// <summary>
    /// 释放资源引用
    /// </summary>
    public void ReleaseResourceReference(string resourceId)
    {
        if (_resourceReferenceCount.TryGetValue(resourceId, out var count))
        {
            count--;
            if (count <= 0)
            {
                _resourceReferenceCount.Remove(resourceId);
            }
            else
            {
                _resourceReferenceCount[resourceId] = count;
            }
        }
    }
    
    /// <summary>
    /// 获取内存状态描述
    /// </summary>
    public MemoryStatus GetStatus()
    {
        return new MemoryStatus
        {
            TotalAllocatedMB = TotalAllocated / (1024 * 1024),
            TotalReservedMB = TotalReserved / (1024 * 1024),
            MonoUsedMB = MonoUsed / (1024 * 1024),
            UsagePercentage = MemoryUsage * 100,
            IsWarning = MemoryUsage >= WARNING_MEMORY_THRESHOLD,
            IsCritical = MemoryUsage >= CRITICAL_MEMORY_THRESHOLD
        };
    }
}

/// 内存状态
/// </summary>
public struct MemoryStatus
{
    public long TotalAllocatedMB;
    public long TotalReservedMB;
    public long MonoUsedMB;
    public float UsagePercentage;
    public bool IsWarning;
    public bool IsCritical;
}

            Destroy(gameObject);
        }
    }
    
    private void Initialize()
    {
        _sourcePool = new Queue<OptimizedAudioSource>();
        _activeSources = new List<OptimizedAudioSource>();
        _clipCache = new Dictionary<string, AudioClip>();
        
        // 预创建音频源
        CreateAudioSourcePool(MAX_SIMULTANEOUS_SOURCES);
    }
    
    /// <summary>
    /// 创建音频源池
    /// </summary>
    private void CreateAudioSourcePool(int count)
    {
        for (int i = 0; i < count; i++)
        {
            var sourceObj = new GameObject($"AudioSource_{i}");
            sourceObj.transform.SetParent(transform);
            
            var audioSource = sourceObj.AddComponent<AudioSource>();
            audioSource.playOnAwake = false;
            audioSource.loop = false;
            audioSource.bypassEffects = true;
            audioSource.bypassListenerEffects = true;
            
            var optimizedSource = new OptimizedAudioSource
            {
                Source = audioSource,
                GameObject = sourceObj,
                IsPlaying = false
            };
            
            sourceObj.SetActive(false);
            _sourcePool.Enqueue(optimizedSource);
        }
    }
    
    /// <summary>
    /// 播放音效（从池中获取）
    /// </summary>
    public void PlaySFX(string clipPath, float volume = 1f, float pitch = 1f)
    {
        if (_sourcePool.Count == 0)
        {
            Debug.LogWarning("No available audio sources");
            return;
        }
        
        var clip = LoadClip(clipPath);
        if (clip == null) return;
        
        var source = _sourcePool.Dequeue();
        
        source.Source.clip = clip;
        source.Source.volume = volume;
        source.Source.pitch = pitch;
        source.Source.Play();
        
        source.IsPlaying = true;
        source.GameObject.SetActive(true);
        
        _activeSources.Add(source);
        
        // 启动协程等待播放完成
        StartCoroutine(WaitForPlayComplete(source));
    }
    
    /// <summary>
    /// 播放循环音乐
    /// </summary>
    public void PlayLoopingMusic(string clipPath, float volume = 0.6f)
    {
        var clip = LoadClip(clipPath);
        if (clip == null) return;
        
        // 找到正在播放的同类型音乐
        var existing = _activeSources.Find(s => s.IsMusic && s.Source.clip == clip);
        if (existing != null)
        {
            existing.Source.volume = volume;
            return;
        }
        
        if (_sourcePool.Count == 0)
        {
            // 需要停止最不重要的音效
            StopLeastImportantSFX();
        }
        
        var source = _sourcePool.Dequeue();
        source.Source.clip = clip;
        source.Source.volume = volume;
        source.Source.loop = true;
        source.Source.Play();
        
        source.IsPlaying = true;
        source.IsMusic = true;
        source.GameObject.SetActive(true);
        
        _activeSources.Add(source);
    }
    
    /// <summary>
    /// 等待播放完成
    /// </summary>
    private System.Collections.IEnumerator WaitForPlayComplete(OptimizedAudioSource source)
    {
        while (source.Source.isPlaying)
        {
            yield return null;
        }
        
        ReturnToPool(source);
    }
    
    /// <summary>
    /// 返回池中
    /// </summary>
    private void ReturnToPool(OptimizedAudioSource source)
    {
        source.IsPlaying = false;
        source.IsMusic = false;
        source.Source.Stop();
        source.Source.clip = null;
        source.GameObject.SetActive(false);
        
        _activeSources.Remove(source);
        _sourcePool.Enqueue(source);
    }
    
    /// <summary>
    /// 停止最不重要的音效
    /// </summary>
    private void StopLeastImportantSFX()
    {
        var nonMusicSources = _activeSources.FindAll(s => !s.IsMusic);
        if (nonMusicSources.Count > 0)
        {
            var toStop = nonMusicSources[0];
            toStop.Source.Stop();
            ReturnToPool(toStop);
        }
    }
    
    /// <summary>
    /// 加载并缓存音频
    /// </summary>
    private AudioClip LoadClip(string path)
    {
        if (_clipCache.TryGetValue(path, out var cached) && cached != null)
        {
            return cached;
        }
        
        var clip = Resources.Load<AudioClip>(path);
        if (clip != null)
        {
            _clipCache[path] = clip;
        }
        return clip;
    }
    
    /// <summary>
    /// 清理缓存
    /// </summary>
    public void ClearCache()
    {
        foreach (var clip in _clipCache.Values)
        {
            if (clip != null)
            {
                Resources.UnloadAsset(clip);
            }
        }
        _clipCache.Clear();
    }
}

/// <summary>

public class OptimizedAudioSource
{
    public AudioSource Source;
    public GameObject GameObject;
    public bool IsPlaying;
    public bool IsMusic;
}

}

    l.log(FATAL, format, args...)
}

// log 内部日志方法
func (l *Logger) log(level LogLevel, format string, args ...interface{}) {
    if level < l.level {
        return
    }
    
    entry := LogEntry{
        ID:        l.id,
        Timestamp: time.Now().UTC(),
        Level:     level,
        Message:   fmt.Sprintf(format, args...),
    }
    
    formatted := l.formatter.Format(entry)
    
    if l.output != nil {
        l.output.WriteString(formatted)
        l.output.WriteString("\n")
    }
    
    // 致命错误退出
    if level == FATAL {
        log.Fatalf(formatted)
    }
}

// SetLevel 设置日志级别
func (l *Logger) SetLevel(level LogLevel) {
    l.level = level
}

// Close 关闭日志
func (l *Logger) Close() {
    if l.output != nil {
        l.output.Close()
    }
}

// LogEntry 日志条目
type LogEntry struct {
    ID        string
    Timestamp time.Time
    Level     LogLevel
    Message   string
}

// Formatter 格式化器接口

    Format(entry LogEntry) string
}

// Format 格式化日志
func (f DefaultFormatter) Format(entry LogEntry) string {
    levelStr := []string{"DEBUG", "INFO", "WARN", "ERROR", "FATAL"}[entry.Level]
    return fmt.Sprintf("[%s] %s %s %s",
        entry.Timestamp.Format(time.RFC3339),
        levelStr,
        entry.ID,
        entry.Message)
}

// JSONFormatter JSON格式化器
type JSONFormatter struct{}

// Format 格式化日志为JSON

    return fmt.Sprintf(`{"timestamp":"%s","level":"%s","id":"%s","message":"%s"}`,
        entry.Timestamp.Format(time.RFC3339),
        []string{"debug", "info", "warning", "error", "fatal"}[entry.Level],

    )

type HealthCheck struct {

// NewHealthCheck 创建健康检查
func NewHealthCheck(db *sql.DB) *HealthCheck {
    return &HealthCheck{db: db}
}

// HealthResponse 健康检查响应
type HealthResponse struct {
    Status    string            `json:"status"`
    Timestamp time.Time         `json:"timestamp"`
    Checks    map[string]Check  `json:"checks"`
}

// Check 单项检查
type Check struct {
    Status  string        `json:"status"`
    Latency time.Duration `json:"latency,omitempty"`
    Error   string        `json:"error,omitempty"`
}

// Check 健康检查
func (h *HealthCheck) Check(c *gin.Context) {
    start := time.Now()
    response := HealthResponse{
        Timestamp: start,
        Checks:    make(map[string]Check),
    }
    
    // 数据库检查
    if err := h.db.Ping(); err != nil {
        response.Checks["database"] = Check{
            Status:  "unhealthy",
            Latency: time.Since(start),
            Error:   err.Error(),
        }
    } else {
        response.Checks["database"] = Check{
            Status:  "healthy",
            Latency: time.Since(start),
        }
    }
    
    // 确定整体状态
    response.Status = "healthy"
    for _, check := range response.Checks {
        if check.Status != "healthy" {
            response.Status = "degraded"
            break
        }
    }
    
    statusCode := http.StatusOK
    if response.Status != "healthy" {
        statusCode = http.StatusServiceUnavailable
    }
    
    c.JSON(statusCode, response)
}

// Readiness 就绪检查
func (h *HealthCheck) Readiness(c *gin.Context) {
    // 检查关键依赖
    if err := h.db.Ping(); err != nil {
        c.JSON(http.StatusServiceUnavailable, gin.H{"ready": false})
        return
    }
    
    c.JSON(http.StatusOK, gin.H{"ready": true})
}

// Liveness 存活检查
func (h *HealthCheck) Liveness(c *gin.Context) {
    c.JSON(http.StatusOK, gin.H{"alive": true})
}

        if (Instance == null)
        {
            Instance = this;
            DontDestroyOnLoad(gameObject);
            
            // 初始化性能设置
            Application.targetFrameRate = (int)TARGET_FRAME_RATE;
            QualitySettings.vSyncCount = 1;
            
            Initialize();
        }
        else
        {
            Destroy(gameObject);
        }
    }
    
    private void Initialize()
    {
        // 按依赖顺序初始化系统
        PerformanceMonitor = new PerformanceMonitor();
        MemoryManager = new MemoryManager();
        DataManager = new DataManager();
        EventManager = new EventManager();
        AudioManager = new AudioManager();
        ProgressManager = new ProgressManager();
        ResourceManager = new ResourceManager();
        RitualEngine = new RitualEngine();
        UIManager = new UIManager();
        
        // 注册性能监控
        PerformanceMonitor.OnPerformanceWarning += HandlePerformanceWarning;
        MemoryManager.OnMemoryWarning += HandleMemoryWarning;
        MemoryManager.OnMemoryCritical += HandleMemoryCritical;
        
        // 订阅事件
        ProgressManager.OnDayComplete += HandleDayComplete;
        RitualEngine.OnRitualComplete += HandleRitualComplete;
        
        CurrentState = GameState.MainMenu;
        
        Debug.Log("[GameManager] All systems initialized");
    }
    
    private void Update()
    {
        Profiler.BeginSample("GameManager.Update");
        
        UpdateFPS();
        PerformanceMonitor.Update();
        MemoryManager.UpdateResources(Time.deltaTime);
        
        Profiler.EndSample();
    }
    
    /// <summary>
    /// FPS 更新
    /// </summary>
    private void UpdateFPS()
    {
        _frameTimeAccumulator += Time.unscaledDeltaTime;
        _frameCount++;
        
        if (Time.unscaledTime - _lastFpsUpdate >= 0.5f)
        {
            CurrentFPS = _frameCount / _frameTimeAccumulator;
            _frameCount = 0;
            _frameTimeAccumulator = 0f;
            _lastFpsUpdate = Time.unscaledTime;
            
            // 性能监控
            PerformanceMonitor.RecordFPS(CurrentFPS);
            
            if (CurrentFPS < 30)
            {
                PerformanceMonitor.OnLowFPS?.Invoke(CurrentFPS);
            }
        }
    }
    
    /// <summary>
    /// 性能警告处理
    /// </summary>
    private void HandlePerformanceWarning(PerformanceLevel level)
    {
        switch (level)
        {
            case PerformanceLevel.Low:
                // 降低质量设置
                QualitySettings.shadowQuality = ShadowQuality.Disable;
                QualitySettings.particlesQuality = ParticleQuality.Low;
                break;
            case PerformanceLevel.Medium:
                // 进一步降低
                QualitySettings.shadowQuality = ShadowQuality.HardOnly;
                break;
            case PerformanceLevel.Critical:
                // 紧急优化
                QualitySettings.shadowQuality = ShadowQuality.Disable;
                QualitySettings.particlesQuality = ParticleQuality.Low;
                QualitySettings.resolutionScalingFixedDPIFactor = 0.8f;
                MemoryManager.PerformStandardCleanup();
                break;
        }
    }
    
    /// <summary>
    /// 内存警告处理
    /// </summary>
    private void HandleMemoryWarning(float usage)
    {
        Debug.LogWarning($"Memory warning: {usage:P1}");
        MemoryManager.PerformStandardCleanup();
    }
    
    /// <summary>
    /// 内存危急处理
    /// </summary>
    private void HandleMemoryCritical(float usage)
    {
        Debug.LogError($"Memory critical: {usage:P1}");
        MemoryManager.PerformAggressiveCleanup();
        
        // 如果在仪式中，可能需要提示用户
        if (CurrentState == GameState.Ritual)
        {
            // 降低音频质量
            AudioManager.SetQuality(AudioQuality.Low);
        }
    }
    
    /// <summary>
    /// 处理每日完成
    /// </summary>
    private void HandleDayComplete(int day)
    {
        EventManager.TrackEvent("day_completed", new Dictionary<string, object>
        {
            { "day", day },
            { "completion_rate", ProgressManager.GetDayCompletionRate(day) },
            { "fps", CurrentFPS }
        });
        
        if (day < 7)
        {
            CurrentDay = day + 1;
            UIManager.ShowDayTransition(day + 1);
        }
        else
        {
            EndSession();
        }
    }
    
    /// <summary>
    /// 处理仪式完成
    /// </summary>
    private void HandleRitualComplete(string ritualType, float completionRate)
    {
        EventManager.TrackEvent("ritual_completed", new Dictionary<string, object>
        {
            { "ritual_type", ritualType },
            { "completion_rate", completionRate },
            { "day", CurrentDay },
            { "memory_usage", MemoryManager.GetStatus().UsagePercentage }
        });
        
        ProgressManager.RecordRitualCompletion(CurrentDay, ritualType, completionRate);
    }
    
    /// <summary>
    /// 开始新会话
    /// </summary>
    public void StartNewSession()
    {
        CurrentSession = new SessionData
        {
            SessionID = Guid.NewGuid().ToString(),
            StartedAt = DateTime.UtcNow,
            DeviceType = SystemInfo.deviceType.ToString(),
            Platform = Application.platform.ToString()
        };
        
        CurrentDay = 1;
        ResourceManager.ResetResources();
        ProgressManager.Reset();
        
        EventManager.TrackEvent("session_started", new Dictionary<string, object>
        {
            { "session_id", CurrentSession.SessionID },
            { "device_type", CurrentSession.DeviceType },
            { "platform", CurrentSession.Platform }
        });
        
        OnSessionStarted?.Invoke();
    }
    
    /// <summary>
    /// 结束会话
    /// </summary>
    public void EndSession()
    {
        CurrentSession.CompletedAt = DateTime.UtcNow;
        CurrentSession.Duration = (CurrentSession.CompletedAt - CurrentSession.StartedAt).TotalSeconds;
        
        DataManager.SubmitSessionData(CurrentSession, () =>
        {
            OnSessionEnded?.Invoke();
            UIManager.ShowSessionSummary();
        });
    }
    
    // 事件
    public event Action OnSessionStarted;
    public event Action OnSessionEnded;
    public event Action<GameState> OnStateChanged;
    
    private void OnDestroy()
    {
        if (Instance == this)
        {
            PerformanceMonitor.OnPerformanceWarning -= HandlePerformanceWarning;
            MemoryManager.OnMemoryWarning -= HandleMemoryWarning;
            MemoryManager.OnMemoryCritical -= HandleMemoryCritical;
            ProgressManager.OnDayComplete -= HandleDayComplete;
            RitualEngine.OnRitualComplete -= HandleRitualComplete;
            
            // 清理资源
            PerformanceMonitor.Dispose();
            MemoryManager.Dispose();
            AudioManager.Dispose();
            EventManager.Dispose();
        }
    }
}

/// <summary>
/// 性能监控器
/// </summary>
public class PerformanceMonitor
{
    private float _fpsThreshold = 45f;
    private float _averageFPS;
    private int _frameCount;
    private float _fpsAccumulator;
    
    public event Action<PerformanceLevel> OnPerformanceWarning;
    public event Action<float> OnLowFPS;
    
    public void Update()
    {
        _frameCount++;
        _fpsAccumulator += Time.unscaledDeltaTime;
        
        if (_fpsAccumulator >= 1f)
        {
            _averageFPS = _frameCount / _fpsAccumulator;
            CheckPerformance();
            
            _frameCount = 0;
            _fpsAccumulator = 0f;
        }
    }
    
    private void CheckPerformance()
    {
        PerformanceLevel level;
        
        if (_averageFPS < 30)
        {
            level = PerformanceLevel.Critical;
        }
        else if (_averageFPS < 45)
        {
            level = PerformanceLevel.Low;
        }
        else if (_averageFPS < 55)
        {
            level = PerformanceLevel.Medium;
        }
        else
        {
            level = PerformanceLevel.Normal;
        }
        
        if (level != PerformanceLevel.Normal)
        {
            OnPerformanceWarning?.Invoke(level);
        }
    }
    
    public void RecordFPS(float fps)
    {
        // 记录到分析系统
    }
    
    public void Dispose()
    {
        OnPerformanceWarning = null;
        OnLowFPS = null;
    }
}

{
    Normal,
    Medium,
    Low,
    Critical
}
